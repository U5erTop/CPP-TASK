## 1. Основная структура программы

```cpp
#include <iostream>         // Подключение библиотеки
```

## 2. Типы данных

### Целые типы

```cpp
int x = 42;                 // Целое число (-2^31 до 2^31-1)
short s = 10;               // Короткое целое (-32768 до 32767)
long l = 1000000L;          // Длинное целое
unsigned int u = 100;       // Беззнаковое целое
```

### Вещественные типы

```cpp
float f = 3.14f;            // Одинарная точность (~6-7 цифр)
double d = 3.14159;         // Двойная точность (~15-16 цифр)
```

### Символьный и логический типы

```cpp
char c = 'A';               // Один символ
bool flag = true;           // Логический тип (true/false)
```

### Размеры типов

```cpp
sizeof(int);                // Возвращает размер типа в байтах
sizeof(x);                  // Размер переменной x
```

## 3. Переменные и константы

### Объявление переменных

```cpp
int a;                      // Объявление без инициализации
int b = 5;                  // Объявление с инициализацией
int c{10};                  // Uniform initialization (C++11)

```

### Константы

```cpp
const double PI = 3.14159;  // Константа (не может быть изменена)
const int MAX = 100;        // Целая константа
#define MAX_SIZE 1000       // Макрос-константа
```

## 4. Операторы

### Арифметические операторы

```cpp
a + b           // Сложение
a - b           // Вычитание
a * b           // Умножение
a / b           // Деление
a % b           // Остаток от деления
a++, ++a        // Инкремент (постфиксный и префиксный)
a--, --a        // Декремент
```

### Составные операторы присваивания

```cpp
a += b;         // Эквивалентно: a = a + b
a -= b;         // Эквивалентно: a = a - b
a *= b;         // Эквивалентно: a = a * b
a /= b;         // Эквивалентно: a = a / b
a %= b;         // Эквивалентно: a = a % b
```

### Операторы сравнения

```cpp
a == b          // Равно
a != b          // Не равно
a < b           // Меньше
a > b           // Больше
a <= b          // Меньше или равно
a >= b          // Больше или равно
```

### Логические операторы

```cpp
a && b          // И (AND) — верно, если оба верны
a || b          // ИЛИ (OR) — верно, если хотя бы одно верно
!a              // НЕ (NOT) — инверсия значения
```

## 5. Ввод и вывод

### Вывод (cout)

```cpp
cout << "Текст";            // Вывод текста
cout << x;                  // Вывод переменной
cout << x << " " << y;      // Вывод нескольких значений
cout << endl;               // Новая строка с очисткой буфера
cout << "\n";               // Новая строка без очистки буфера
cout << "x = " << x << endl;// Комбинированный вывод
```

### Ввод (cin)

```cpp
cin >> x;                   // Ввод в переменную
cin >> x >> y;              // Ввод нескольких значений
cin >> str;                 // Ввод строки (без пробелов)
getline(cin, str);          // Ввод строки со смыслом (с пробелами)
```

## 6. Условные операторы

### Оператор if

```cpp
if (condition) {
    // Код если condition == true
}
```

### Оператор if...else

```cpp
if (condition) {
    // Код если true
} else {
    // Код если false
}
```

### Оператор if...else if...else

```cpp
if (condition1) {
    // ...
} else if (condition2) {
    // ...
} else if (condition3) {
    // ...
} else {
    // ...
}
```

### Оператор switch

```cpp
switch (expression) {
    case value1:
        // Код для value1
        break;
    case value2:
        // Код для value2
        break;
    default:
        // Код по умолчанию
}
```

### Тернарный оператор

```cpp
condition ? value1 : value2;   // Возвращает value1 если true, иначе value2
int max = (a > b) ? a : b;
```

## 7. Циклы

### Цикл while

```cpp
while (condition) {
    // Код выполняется пока condition == true
}
```

### Цикл do...while

```cpp
do {
    // Код выполняется хотя бы один раз
} while (condition);
```

### Цикл for

```cpp
for (int i = 0; i < n; i++) {
    // i = начальное значение
    // i < n = условие продолжения
    // i++ = изменение на каждой итерации
}
```

### Диапазонный цикл (C++11)

```cpp
for (int x : array) {
    // x принимает каждое значение из array
}
```

### Управление циклом

```cpp
break;              // Выход из цикла
continue;           // Переход к следующей итерации
```

## 8. Функции

### Определение функции

```cpp
return_type function_name(parameter_type param1, parameter_type param2) {
    // Тело функции
    return value;   // Возврат результата (если return_type не void)
}
```

### Примеры функций

```cpp
// Функция без параметров
void printHello() {
    cout << "Hello!" << endl;
}

// Функция с параметрами
int add(int a, int b) {
    return a + b;
}

// Функция без возврата
void greet(string name) {
    cout << "Hello, " << name << "!" << endl;
}

// Функция с параметром по умолчанию
void print(int num = 0) {
    cout << "Number: " << num << endl;
}
```

### Вызов функции

```cpp
printHello();           // Вызов функции без параметров
int result = add(5, 3); // Вызов функции с параметрами
greet("John");          // Передача строки как параметра
```

## 9. Строки

### Работа со строками (string)

```cpp
#include <string>
using namespace std;

string str = "Hello";       // Объявление строки
string s1 = "World";
string s2 = s1;             // Копирование

// Конкатенация
string s3 = str + " " + s1;

// Методы строк
str.length();               // Длина строки
str.size();                 // Размер строки (то же самое)
str[0];                     // Доступ к символу
str.at(0);                  // Безопасный доступ
str.substr(0, 5);           // Подстрока
str.find("ll");             // Поиск подстроки
str.replace(0, 1, "J");     // Замена

// Сравнение
str == s1;                  // Сравнение строк
str != s1;
str < s1;
```

### Считывание строк

```cpp
string line;
cin >> line;                // Считывание одного слова
getline(cin, line);         // Считывание всей строки (с пробелами)
```

## 10. Массивы

### Одномерные массивы

```cpp
int arr[5];                 // Объявление массива из 5 элементов
int arr[5] = {1, 2, 3, 4, 5};  // С инициализацией
int arr[] = {1, 2, 3};      // Размер определяется автоматически

arr[0];                     // Доступ к первому элементу
arr[4];                     // Доступ к пятому элементу

// Проход по массиву
for (int i = 0; i < 5; i++) {
    cout << arr[i];
}

// Диапазонный цикл
for (int x : arr) {
    cout << x;
}
```

### Двумерные массивы

```cpp
int matrix[3][3];          // Матрица 3×3
int matrix[2][3] = {
    {1, 2, 3},
    {4, 5, 6}
};

matrix[0][0];              // Доступ к элементу
```

## 15. Библиотеки

### Основные библиотеки

```cpp
#include <iostream>         // Ввод-вывод
#include <string>           // Работа со строками
#include <cmath>            // Математические функции
#include <cstdlib>          // Стандартные функции (rand, abs)
#include <vector>           // Динамические массивы
#include <algorithm>        // Алгоритмы (sort, find и т.д.)
#include <fstream>          // Работа с файлами
```

### Часто используемые функции

```cpp
// Математика
abs(x);                     // Абсолютное значение
sqrt(x);                    // Квадратный корень
pow(x, y);                  // x в степени y
sin(x), cos(x), tan(x);     // Тригонометрические функции
floor(x), ceil(x);          // Округление вниз/вверх
rand();                     // Случайное число

// Строки
str.length();               // Длина строки
str.find(substr);           // Поиск подстроки
str.substr(pos, len);       // Подстрока
tolower(c);                 // В нижний регистр
toupper(c);                 // В верхний регистр
```

## 16. Обработка ошибок

### Проверка ввода

```cpp
if (!(cin >> x)) {
    cout << "Ошибка ввода!" << endl;
    cin.clear();            // Очистка флага ошибки
    cin.ignore();           // Пропуск остатка строки
}
```

### Проверка деления на ноль

```cpp
if (b != 0) {
    result = a / b;
} else {
    cout << "Ошибка: деление на ноль!" << endl;
}
```
